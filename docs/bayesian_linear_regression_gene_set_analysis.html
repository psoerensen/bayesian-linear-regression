Introduction

This document demonstrates Bayesian Linear Regression using conjugate Gaussian and scaled inverse-chi-squared priors. The model and priors follow the theory outlined in the notes above. We implement a Gibbs sampler for inference, summarize posterior distributions, and assess convergence diagnostics.

Model Setup and Data Simulation

Code 
library(qgg)
# ---- Step 1: Basic parameters ----
n_genes <- 20000
n_pathways <- 10
genes <- paste0("Gene", seq_len(n_genes))

# ---- Step 2: Simulate gene-level association statistics ----
# Null genes: Z ~ N(0,1)
z_scores <- rnorm(n_genes, mean = 0, sd = 1)
names(z_scores) <- genes

# A subset of truly associated genes
n_assoc <- 100
associated_genes <- sample(genes, n_assoc)

# Associated genes have stronger effects
z_scores[associated_genes] <- rnorm(n_assoc, mean = 3, sd = 1)

# Compute two-sided p-values
p_values <- 2 * pnorm(-abs(z_scores))
gene_data <- data.frame(Gene = genes, Z = z_scores, P = p_values)

# ---- Step 3: Draw realistic gene set sizes ----
# Log-normal distribution gives right-skewed sizes (median ≈ 80)
sizes_raw <- round(rlnorm(n_pathways, meanlog = log(80), sdlog = 0.6))
sizes <- pmin(pmax(sizes_raw, 15), 500)  # truncate to 15–500

# ---- Step 4: Create overlapping pathways ----
pathway_list <- vector("list", n_pathways)
names(pathway_list) <- paste0("Pathway_", seq_len(n_pathways))

for (i in seq_len(n_pathways)) {
  size <- sizes[i]
  
  if (i > 1 && runif(1) < 0.9) {  # 90% chance of overlap with earlier sets
    j <- sample(seq_len(i - 1), 1)  # choose an existing pathway
    target_overlap <- round(size * runif(1, 0.20, 0.50))  # 20–50% overlap
    
    overlap_pool <- pathway_list[[j]]
    k <- min(target_overlap, length(overlap_pool), size)
    overlap_genes <- if (k > 0) sample(overlap_pool, k) else character(0)
    
    # Add remaining genes from global pool (non-overlapping)
    remaining_pool <- setdiff(genes, overlap_genes)
    n_remaining <- size - length(overlap_genes)
    new_genes <- if (n_remaining > 0) sample(remaining_pool, n_remaining) else character(0)
    
    pathway_list[[i]] <- unique(c(overlap_genes, new_genes))
  } else {
    # Independent pathway (no enforced overlap)
    pathway_list[[i]] <- sample(genes, size)
  }
}

# ---- Step 5: Enrich 1–2 pathways ----
n_enriched_paths <- 2
enriched_paths <- sample(names(pathway_list), n_enriched_paths)

for (p in enriched_paths) {
  n_enriched_genes <- 30
  enriched_genes <- sample(associated_genes, n_enriched_genes)
  pathway_list[[p]] <- unique(c(pathway_list[[p]], enriched_genes))
}

X <- scale(X, center = TRUE, scale = FALSE)
n <- length(y)

p <- ncol(X) 
beta_true <- rep(0, p)

Gibbs Sampler Implementation

We use conjugate priors:

  • \(\beta \mid \sigma^2, \sigma_b^2 \sim \mathcal{N}(0, \sigma_b^2 I_p)\)

  • \(\sigma_b^2 \sim S_b \chi^{-2}(v_b)\)

  • \(\sigma^2 \sim S \chi^{-2}(v)\)

Code 
# Hyperparameters
v_b <- 4
S_b <- 1
v <- 4
S <- 1

# Initialization
beta <- rep(0, ncol(X))
sigma2 <- 1
sigma2_b <- 1

n_iter <- 5000
burn_in <- 1000

# Store samples
beta_samples <- matrix(NA, n_iter, ncol(X))
sigma2_samples <- numeric(n_iter)
sigma2_b_samples <- numeric(n_iter)

for (t in 1:n_iter) {
  # Sample beta | sigma2, sigma2_b, y
  Sigma_beta <- solve(t(X) %*% X / sigma2 + diag(1 / sigma2_b, ncol(X)))
  mu_beta <- Sigma_beta %*% t(X) %*% y / sigma2
  beta <- as.numeric(mu_beta + t(chol(Sigma_beta)) %*% rnorm(ncol(X)))
  
  # Sample sigma_b^2 | beta
  v_b_tilde <- v_b + ncol(X)
  S_b_tilde <- (sum(beta^2) + v_b * S_b) / v_b_tilde
  sigma2_b <- v_b_tilde * S_b_tilde / rchisq(1, df = v_b_tilde)
  
  # Sample sigma^2 | beta, y
  v_tilde <- v + n
  resid <- y - X %*% beta
  S_tilde <- (sum(resid^2) + v * S) / v_tilde
  sigma2 <- v_tilde * S_tilde / rchisq(1, df = v_tilde)
  
  # Store samples
  beta_samples[t, ] <- beta
  sigma2_samples[t] <- sigma2
  sigma2_b_samples[t] <- sigma2_b
}

posterior_summary <- function(samples, probs = c(0.025, 0.5, 0.975)) {
  c(
    mean = mean(samples),
    sd = sd(samples),
    quantile(samples, probs = probs)
  )
}

Convergence Diagnostics

We compute autocorrelation, Monte Carlo standard error, Geweke Z-score, Gelman-Rubin (), and effective sample size.

Code 
# Simple diagnostics for one chain
convergence_stats <- function(samples) {
  n <- length(samples)
  ac1 <- cor(samples[-1], samples[-n])
  mcse <- sd(samples) * sqrt((1 + ac1) / n)
  a <- floor(0.1 * n); b <- floor(0.5 * n)
  z <- (mean(samples[1:a]) - mean(samples[(n - b + 1):n])) /
       sqrt(var(samples[1:a]) / a + var(samples[(n - b + 1):n]) / b)
  ess <- n / (1 + 2 * sum(acf(samples, plot = FALSE)$acf[-1]))
  c(autocorr1 = ac1, mcse = mcse, geweke_z = z, ess = ess)
}
